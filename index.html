Practical 4 – Hypothesis Testing

Aim: Perform hypothesis testing using two-sample t-test.

import numpy as np
from scipy import stats
import matplotlib.pyplot as plt

np.random.seed(42)

sample1 = np.random.normal(10,2,30)
sample2 = np.random.normal(12,2,30)

t,p = stats.ttest_ind(sample1,sample2)

print("T-statistic:",t)
print("P-value:",p)

plt.hist(sample1,alpha=0.5)
plt.hist(sample2,alpha=0.5)

plt.show()
